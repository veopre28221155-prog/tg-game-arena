function startEmu(file, dummy, rid, isSolo) { let core = 'segaMD'; if(file.endsWith('.nes')) core = 'nes'; if(file.endsWith('.sfc')) core = 'snes'; document.getElementById('app').style.display = 'none'; document.getElementById('emulator-layer').style.display = 'block'; document.getElementById('ejs-container').innerHTML = ""; // Глобальные настройки эмулятора window.EJS_player = '#ejs-container'; window.EJS_core = core; window.EJS_gameUrl = window.location.origin + '/' + file; window.EJS_pathtodata = window.location.origin + '/data/'; window.EJS_startOnLoaded = true; window.EJS_virtualGamepad = true; window.EJS_onGameStart = () => { console.log("SYNC START"); // Принудительный перезапуск для синхронизации кадров if(EJS_emulator.gameManager) EJS_emulator.gameManager.restart(); setInterval(() => { if (typeof EJS_emulator !== 'undefined') { const ram = EJS_emulator.gameManager.getWRam(); score = file.includes('sonic') ? ram[0xFE20] : (file.includes('umk3') ? ram[0xB110] : 0); document.getElementById('emu-score').innerText = score; } }, 1000); }; if(!isSolo) { // ОПРЕДЕЛЕНИЕ РОЛИ const isHost = user.telegramId == lobby.creatorId; const myPort = isHost ? 0 : 1; const partnerPort = isHost ? 1 : 0; console.log(`Role: ${isHost ? 'HOST' : 'GUEST'} | My Port: ${myPort}`); // ОТПРАВКА СВОИХ КНОПОК const sendInput = (e, type) => { socket.emit('emu-input', { rid, key: e.keyCode, type: type, pNum: myPort // Передаем, какой порт мы нажали реально }); }; window.onkeydown = (e) => sendInput(e, 'keydown'); window.onkeyup = (e) => sendInput(e, 'keyup'); // ПРИЕМ КНОПОК ПАРТНЕРА socket.on('partner-input', (d) => { if(typeof EJS_emulator !== 'undefined' && EJS_emulator.gameManager) { // d.pNum - это порт, который нажал друг. // Мы эмулируем его именно в тот же порт у себя. EJS_emulator.gameManager.simulateInput(d.key, d.type, d.pNum); } }); } const s = document.createElement('script'); s.src = window.location.origin + '/data/loader.js?v=' + Date.now(); document.body.appendChild(s); }
