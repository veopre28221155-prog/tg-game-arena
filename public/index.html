function startEmu(file, coreType, rid, isSolo) { document.getElementById('app').style.display = 'none'; document.getElementById('emulator-layer').style.display = 'block'; document.getElementById('ejs-container').innerHTML = ""; // 1. Определение ролей const isHost = !isSolo && (user.telegramId == lobby.creatorId); const myPort = isHost ? 0 : 1; // 2. Глобальные настройки window.EJS_player = '#ejs-container'; window.EJS_core = file.endsWith('.nes') ? 'nes' : 'segaMD'; window.EJS_gameUrl = window.location.origin + '/' + file; window.EJS_pathtodata = window.location.origin + '/data/'; window.EJS_startOnLoaded = true; window.EJS_virtualGamepad = true; // 3. Синхронизация при старте window.EJS_onGameStart = () => { console.log("System Status: Frame Zero Syncing..."); if (EJS_emulator.gameManager) { // Принудительный рестарт для выравнивания кадров EJS_emulator.gameManager.restart(); // 4. Автоматический вход в режим 2-Player (Macro Script) if (!isSolo) { setTimeout(() => { console.log("Macro: Selecting 2-Player Mode..."); // Эмулируем нажатие ВНИЗ (код 40) на джойстике Хоста у обоих EJS_emulator.gameManager.simulateInput(40, 'keydown', 0); setTimeout(() => { EJS_emulator.gameManager.simulateInput(40, 'keyup', 0); // Эмулируем нажатие START (код 13) через полсекунды setTimeout(() => { EJS_emulator.gameManager.simulateInput(13, 'keydown', 0); setTimeout(() => EJS_emulator.gameManager.simulateInput(13, 'keyup', 0), 100); console.log("Macro: Match Initiated."); }, 500); }, 100); }, 4000); // 4 секунды на прогрузку заставки BIOS/Sega } } // Мониторинг очков setInterval(() => { if (typeof EJS_emulator !== 'undefined') { const ram = EJS_emulator.gameManager.getWRam(); score = file.includes('sonic') ? ram[0xFE20] : (file.includes('umk3') ? ram[0xB110] : 0); document.getElementById('emu-score').innerText = score; } }, 1000); }; // 5. Мост ввода (Network Bridge) if (!isSolo) { // Отправка своего ввода const handleLocalInput = (e, type) => { socket.emit('emu-input', { rid, key: e.keyCode, type: type, pNum: myPort }); }; window.onkeydown = (e) => handleLocalInput(e, 'keydown'); window.onkeyup = (e) => handleLocalInput(e, 'keyup'); // Инъекция ввода партнера socket.on('partner-input', (d) => { if (typeof EJS_emulator !== 'undefined' && EJS_emulator.gameManager) { // Прямая инъекция в порт партнера EJS_emulator.gameManager.simulateInput(d.key, d.type, d.pNum); } }); } // Загрузка лоадера const s = document.createElement('script'); s.src = window.location.origin + '/data/loader.js?v=' + Date.now(); document.body.appendChild(s); }
