function launchGame(t) { gameType = t; score = 0; navScreen('s-game'); // Очистка старого эмулятора, если он был const oldEmu = document.getElementById('sonic-emulator'); if (oldEmu) oldEmu.remove(); const canvas = document.getElementById('gameCanvas'); const controls = document.getElementById('game-controls'); if (t === 'sonic') { canvas.style.display = 'none'; controls.style.display = 'none'; // У эмулятора свои кнопки const ifr = document.createElement('iframe'); ifr.id = 'sonic-emulator'; ifr.allow = 'autoplay; fullscreen'; // Формируем АБСОЛЮТНЫЙ путь к рому const romUrl = window.location.origin + '/roms/sonic.bin'; // Используем официальный загрузчик EmulatorJS ifr.src = `https://emulatorjs.org/emu/index.html?game=${encodeURIComponent(romUrl)}&core=genesis`; // Вставляем фрейм canvas.parentNode.insertBefore(ifr, canvas.nextSibling); sonicStartTime = Date.now(); if(loop) clearInterval(loop); loop = setInterval(() => { score = Date.now() - sonicStartTime; document.getElementById('g-score').innerText = score + " ms"; }, 100); } else { canvas.style.display = 'block'; controls.style.display = 'flex'; // Тут запуск змейки или других игр runSnake(); } }
