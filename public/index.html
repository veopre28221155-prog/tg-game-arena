let scoreInterval = null; let syncInterval = null; function startEmu(file, core, rid, isSolo) { document.getElementById('app').style.display = 'none'; document.getElementById('emulator-layer').style.display = 'block'; document.getElementById('ejs-container').innerHTML = ""; window.EJS_player = '#ejs-container'; window.EJS_core = core; window.EJS_gameUrl = window.location.origin + '/' + file; window.EJS_pathtodata = window.location.origin + '/data/'; window.EJS_startOnLoaded = true; window.EJS_virtualGamepad = true; window.EJS_onGameStart = () => { const gm = EJS_emulator.gameManager; if (!gm) return; if (isSolo) { gm.restart(); } else { // 1. НЕМЕДЛЕННАЯ ПАУЗА ДЛЯ ОЖИДАНИЯ ПАРТНЕРА gm.pause(); document.getElementById('sync-indicator').innerText = "[WAITING FOR PARTNER...]"; socket.emit('client-loaded', { rid }); // 2. СИГНАЛ ЗАЖИГАНИЯ ОТ СЕРВЕРА socket.off('ignite-engine'); socket.on('ignite-engine', () => { console.log("IGNITION: Starting synchronous session"); document.getElementById('sync-indicator').innerText = "[ENGINE IGNITED]"; gm.restart(); // Обнуляем циклы CPU // 3. АВТО-МЕНЮ МАКРОС (Выполняется синхронно) setTimeout(() => { gm.simulateInput(40, 'keydown', 0); // Down setTimeout(() => { gm.simulateInput(40, 'keyup', 0); setTimeout(() => { gm.simulateInput(13, 'keydown', 0); // Start setTimeout(() => gm.simulateInput(13, 'keyup', 0), 100); }, 500); }, 100); }, 3000); // 4. ВКЛЮЧЕНИЕ SNAPSHOT SYNC (Только для Хоста) const isHost = user.telegramId == lobby.creatorId; if (isHost) { setTimeout(() => { syncInterval = setInterval(async () => { const state = await gm.getState(); socket.emit('sync-state', { rid, state }); }, 5000); }, 10000); } }); // 5. ПРИМЕНЕНИЕ SNAPSHOT (Для Гостя) socket.off('apply-state'); socket.on('apply-state', (state) => { gm.loadState(state); document.getElementById('sync-indicator').innerText = "[RAM SYNC: OK]"; }); // 6. ОБМЕН КНОПКАМИ const myPort = (user.telegramId == lobby.creatorId) ? 0 : 1; window.onkeydown = (e) => socket.emit('emu-input', { rid, key: e.keyCode, type: 'keydown', pNum: myPort }); window.onkeyup = (e) => socket.emit('emu-input', { rid, key: e.keyCode, type: 'keyup', pNum: myPort }); socket.off('partner-input'); socket.on('partner-input', (d) => gm.simulateInput(d.key, d.type, d.pNum)); } // Трекинг очков scoreInterval = setInterval(() => { if (typeof EJS_emulator !== 'undefined') { const ram = gm.getWRam(); score = file.includes('sonic') ? ram[0xFE20] : (file.includes('umk3') ? ram[0xB110] : 0); document.getElementById('emu-score').innerText = score || 0; } }, 1000); }; const s = document.createElement('script'); s.src = window.location.origin + '/data/loader.js?v=' + Date.now(); document.body.appendChild(s); }
