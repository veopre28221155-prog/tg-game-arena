let syncInterval = null; // Глобальная переменная для цикла синхронизации function startEmu(file, coreType, rid, isSolo) { if (window.EJS_emulator) { try { window.EJS_emulator.stop(); } catch(e){} } if (scoreInterval) clearInterval(scoreInterval); if (syncInterval) clearInterval(syncInterval); document.getElementById('app').style.display = 'none'; document.getElementById('emulator-layer').style.display = 'block'; document.getElementById('ejs-container').innerHTML = ""; const isHost = !isSolo && (user.telegramId == lobby.creatorId); const myPort = isHost ? 0 : 1; window.EJS_player = '#ejs-container'; window.EJS_core = file.endsWith('.nes') ? 'nes' : 'segaMD'; window.EJS_gameUrl = window.location.origin + '/' + file; window.EJS_pathtodata = window.location.origin + '/data/'; window.EJS_startOnLoaded = true; window.EJS_virtualGamepad = true; window.EJS_onGameStart = () => { if (EJS_emulator.gameManager) { EJS_emulator.gameManager.restart(); if (!isSolo) { // --- ЛОГИКА ТЯЖЕЛОЙ СИНХРОНИЗАЦИИ --- if (isHost) { // ХОСТ: Каждые 5 секунд делает снимок и шлет Гостю syncInterval = setInterval(async () => { if (EJS_emulator.gameManager) { const state = await EJS_emulator.gameManager.getState(); socket.emit('sync-state', { rid: rid, state: state }); console.log("[Sync] State snapshot sent to partner"); } }, 5000); } else { // ГОСТЬ: Слушает снимки от Хоста и применяет их socket.off('apply-state'); socket.on('apply-state', async (binaryState) => { if (EJS_emulator.gameManager) { await EJS_emulator.gameManager.loadState(binaryState); console.log("[Sync] State applied from host"); } }); } // Авто-меню (2 Players) setTimeout(() => { EJS_emulator.gameManager.simulateInput(40, 'keydown', 0); setTimeout(() => { EJS_emulator.gameManager.simulateInput(40, 'keyup', 0); setTimeout(() => { EJS_emulator.gameManager.simulateInput(13, 'keydown', 0); setTimeout(() => EJS_emulator.gameManager.simulateInput(13, 'keyup', 0), 100); }, 500); }, 100); }, 4000); } } scoreInterval = setInterval(() => { if (typeof EJS_emulator !== 'undefined') { const ram = EJS_emulator.gameManager.getWRam(); score = file.includes('sonic') ? ram[0xFE20] : (file.includes('umk3') ? ram[0xB110] : 0); document.getElementById('emu-score').innerText = score || 0; } }, 1000); }; if(!isSolo) { window.onkeydown = (e) => socket.emit('emu-input', { rid, key: e.keyCode, type: 'keydown', pNum: myPort }); window.onkeyup = (e) => socket.emit('emu-input', { rid, key: e.keyCode, type: 'keyup', pNum: myPort }); socket.off('partner-input'); socket.on('partner-input', (d) => { if(typeof EJS_emulator !== 'undefined' && EJS_emulator.gameManager) { EJS_emulator.gameManager.simulateInput(d.key, d.type, d.pNum); } }); } const s = document.createElement('script'); s.src = window.location.origin + '/data/loader.js'; document.body.appendChild(s); }
