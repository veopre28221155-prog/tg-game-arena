async function init() { // FRONTEND FIREWALL if (!tg.initData || tg.initData === "") { return tg.showAlert("CRITICAL: Open app via Telegram Mobile!"); } try { const response = await fetch('/api/user-data', { method: 'POST', headers: {'Content-Type': 'application/json'}, body: JSON.stringify({initData: tg.initData}) }); const r = await response.json(); if(!r.success) return debugOut(r); user = r.user; document.getElementById('p-bal').innerText = user.balance; const saved = sessionStorage.getItem('currentLid'); if (saved) socket.emit('reconnect-to-slot', { rid: saved, uid: user.telegramId }); const sp = tg.initDataUnsafe.start_param; if (sp && sp.startsWith('lobby_')) joinLobby(sp.replace('lobby_','')); } catch (err) { debugOut({ error: "Network or Server offline", stack: "INIT_CATCH" }); } }
