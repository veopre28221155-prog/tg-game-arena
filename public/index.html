// Добавь эту переменную в самый верх скрипта, где у тебя user и lobby let scoreInterval = null; function startEmu(file, coreType, rid, isSolo) { // 0. Полная очистка контейнера и интервалов if (window.EJS_emulator) { try { window.EJS_emulator.stop(); } catch(e) {} } if (scoreInterval) clearInterval(scoreInterval); document.getElementById('app').style.display = 'none'; document.getElementById('emulator-layer').style.display = 'block'; document.getElementById('ejs-container').innerHTML = ""; // 1. Определение ролей const isHost = !isSolo && (user.telegramId == lobby.creatorId); const myPort = isHost ? 0 : 1; // 2. Глобальные настройки EmulatorJS window.EJS_player = '#ejs-container'; window.EJS_core = file.endsWith('.nes') ? 'nes' : 'segaMD'; window.EJS_gameUrl = window.location.origin + '/' + file; window.EJS_pathtodata = window.location.origin + '/data/'; window.EJS_startOnLoaded = true; window.EJS_virtualGamepad = true; // 3. Автоматизация старта и синхронизация циклов window.EJS_onGameStart = () => { console.log("Macro-Sync: Atomic Reset Executed."); if (EJS_emulator.gameManager) { // ПРИНУДИТЕЛЬНЫЙ РЕСТАРТ ОБОИХ ЯДЕР EJS_emulator.gameManager.restart(); // 4. Скрипт автоматического входа в режим 2-Player if (!isSolo) { setTimeout(() => { console.log("Macro-Script: Navigating to 2P mode..."); // Имитируем выбор в меню (ВНИЗ на Порту 0 у обоих) EJS_emulator.gameManager.simulateInput(40, 'keydown', 0); setTimeout(() => { EJS_emulator.gameManager.simulateInput(40, 'keyup', 0); // Нажимаем START на Порту 0 setTimeout(() => { EJS_emulator.gameManager.simulateInput(13, 'keydown', 0); setTimeout(() => EJS_emulator.gameManager.simulateInput(13, 'keyup', 0), 100); }, 500); }, 100); }, 4000); // Тайминг ожидания заставки Sega/Nintendo } } // 5. Мониторинг оперативной памяти (Score Tracking) scoreInterval = setInterval(() => { if (typeof EJS_emulator !== 'undefined' && EJS_emulator.gameManager) { const ram = EJS_emulator.gameManager.getWRam(); if (file.includes('sonic')) score = ram[0xFE20]; else if (file.includes('umk3')) score = ram[0xB110]; document.getElementById('emu-score').innerText = score || 0; } }, 1000); }; // 6. Сетевой Мост Ввода (Network Input Relay) if (!isSolo) { const sendInput = (e, type) => { socket.emit('emu-input', { rid: rid, key: e.keyCode, type: type, pNum: myPort }); }; window.onkeydown = (e) => sendInput(e, 'keydown'); window.onkeyup = (e) => sendInput(e, 'keyup'); socket.off('partner-input'); socket.on('partner-input', (d) => { if (typeof EJS_emulator !== 'undefined' && EJS_emulator.gameManager) { EJS_emulator.gameManager.simulateInput(d.key, d.type, d.pNum); } }); } const s = document.createElement('script'); s.src = window.location.origin + '/data/loader.js?v=' + Date.now(); document.body.appendChild(s); }
